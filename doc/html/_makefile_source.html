1# /**
2# * @file Makefile
3# * @brief Fichier de compilation du projet.
4# *
5# * Ce fichier Makefile gère la compilation, l'exécution et les tests du projet.
6# * Il organise la compilation des fichiers sources, la gestion des tests et le nettoyage des fichiers générés.
7# *
8# * @details
9# * - `make all` : Compile le programme principal.
10# * - `make run` : Exécute le programme principal.
11# * - `make test` : Compile et exécute les tests.
12# * - `make clean` : Supprime les fichiers objets et les exécutables.
13# */
14
15# Définition des variables
16CXX = g++
17CXXFLAGS = -Wall -Wextra -std=c++11 -Iinclude
18SRC_DIR = src
19TEST_DIR = tests
20OBJ_DIR = obj
21DATA_DIR = data
22BIN_DIR = bin
23EXEC = $(BIN_DIR)/main
24TEST_RESULTS = $(TEST_DIR)/results/test_results.txt
25
26# Fichiers source et objets
27SRCS = $(wildcard $(SRC_DIR)/*.cpp)
28OBJS = $(patsubst %.cpp, $(OBJ_DIR)/%.o, $(SRCS))
29
30# Fichiers de test et exécutables de test
31TEST_SRCS = $(wildcard $(TEST_DIR)/*.cpp)
32TEST_EXECS = $(patsubst $(TEST_DIR)/%.cpp, $(BIN_DIR)/%, $(TEST_SRCS))
33TEST_OBJS = $(patsubst %.cpp, $(OBJ_DIR)/%.o, $(TEST_SRCS))
34
35# Création des répertoires avant compilation
36create_dirs:
37 if not exist $(OBJ_DIR) mkdir $(OBJ_DIR)
38 if not exist $(OBJ_DIR)\src mkdir $(OBJ_DIR)\src
39 if not exist $(BIN_DIR) mkdir $(BIN_DIR)
40 if not exist $(DATA_DIR) mkdir $(DATA_DIR)
41 if not exist $(TEST_DIR)\results mkdir $(TEST_DIR)\results
42
43# Compilation du programme principal
44all: create_dirs $(EXEC)
45
46$(EXEC): $(OBJS) obj/main.o
47 $(CXX) $(CXXFLAGS) $^ -o $@
48
49# Compilation des fichiers source en objets
50$(OBJ_DIR)/%.o: %.cpp | create_dirs
51 $(CXX) $(CXXFLAGS) -c $< -o $@
52
53# Exécution du programme principal
54run: all
55 @echo "Execution du programme principal..."
56 @$(EXEC)
57
58# Compilation des tests
59$(BIN_DIR)/%: $(TEST_DIR)/%.cpp $(OBJS) | create_dirs
60 $(CXX) $(CXXFLAGS) $< $(OBJS) -o $@
61
62# Exécution des tests et sauvegarde des résultats dans un fichier test_results.txt
63test: $(TEST_EXECS)
64 @echo "Execution des tests..."
65 if exist $(TEST_RESULTS) del $(TEST_RESULTS)
66 @for %%t in ($(TEST_EXECS)) do ( \
67 echo test %%~nt >> $(TEST_RESULTS) & \
68 $(BIN_DIR)\%%~nt.exe >> $(TEST_RESULTS) 2>&1 & \
69 echo. >> $(TEST_RESULTS) \
70 )
71 @echo "Les resultats des tests sont sauvegardes dans $(TEST_RESULTS)."
72
73# Nettoyage des fichiers objets et exécutables
74clean:
75 if exist $(OBJ_DIR) rmdir /S /Q $(OBJ_DIR)
76 if exist $(BIN_DIR) rmdir /S /Q $(BIN_DIR)
77 if exist $(TEST_DIR)\results rmdir /S /Q $(TEST_DIR)\results
78 if exist $(DATA_DIR)\rapport_analyse.csv del $(DATA_DIR)\rapport_analyse.csv
79 if exist $(TEST_RESULTS) del $(TEST_RESULTS)
